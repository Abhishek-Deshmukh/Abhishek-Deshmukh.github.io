Code for air-drag

def reflect(self,slope_of_surface):
    # slope of deflected_to=tan(2*atan(slope_of_surface))
    deflected_to = math.tan(2*math.atan(slope_of_surface))

    # assuming final speed to be the same. coz that would be the average effect
    final_velocity_x = self.current_velocity_x / math.sqrt(1 + deflected_to*deflected_to)
    final_velocity_y = deflected_to * final_velocity_x

    # updating momentum changes
    self.momentum_x += mass * (self.current_velocity_x - final_velocity_x)
    self.momentum_y += mass * (self.current_velocity_y - final_velocity_y)

    # setting up for another collision
    self.current_velocity_x = final_velocity_x
    self.current_velocity_y = final_velocity_y
